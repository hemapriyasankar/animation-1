# animation-1
import turtle as t


t.speed(0)


t.tracer(3)


t.setpos(0,0)


t.bgcolor('black')


t.hideturtle()


color=('red','cyan','white')


for i in range (400):
	
	
	t.pencolor(color[i%3])
	
	
	t.pensize(2)
	
	
	t.fd(i)
	
	
	t.rt(120)
	
	
	for j in range(3):
		
		
		t.fd(i)
		
		
		t.rt(120)
		
		
		for k in range(2):
			
			
			t.fd(i)
			
			
			t.lt(10)
			
			
			for l in range(1):
				
				
				t.fd(i)
				
				
				t.rt(90)
				
				
t.exitonclick
